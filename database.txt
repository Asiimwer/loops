        function grade($mark){
        $grade="";
        if($mark>=90 && $mark<=100) {
          $grade="D1";
        }elseif($mark>=80 && $mark<=89){
          $grade="D2";
        }elseif($mark>=70 && $mark<=79){
          $grade="C3";
        }elseif($mark>=60 && $mark<=69){
          $grade="C4";
        }elseif($mark>=50 && $mark<=59){
          $grade="C5";
        }elseif($mark>=40 && $mark<=49){
          $grade="C6";
        }elseif($mark>=30 && $mark<=39){
          $grade="P7";
        }elseif($mark>=20 && $mark<=29){
          $grade="P8";
        }elseif($mark>=0 && $mark<=19){
          $grade="F9";
        }
        return $grade;
      }
